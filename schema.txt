Table index{
  index_id int [primary key]
  symbol text
}
Table industry{
  industry_id int [primary key]
  symbol text
}
Table sector{
  sector_id int [primary key]
  symbol text
  industry_id int [ref: > industry.industry_id]
}
Table location{
  location_id int [primary key]
  location_name text
  lat real
  lon real
}

  symbol text
  name text
}
Table crypto_price_day{
  price_id int [primary key]
  crypto_id int [ref: > crypto.crypto_id]
  datetime text
  open real
  high real
  low real
  close real
  volume int
}

  price_id int [primary key]

  high real
  low real
  close real

  datetime text
  link text
  content text
}
Table many_crypto_news{
  news_id int [ref: > crypto_news.news_id]
  crypto_id int [ref: > crypto.crypto_id]

Table crypto_location{
  news_id int [ref: > crypto_news.news_id]
  crypto_id int [ref: > crypto.crypto_id]
  location_id int [ref: > location.location_id]
}
